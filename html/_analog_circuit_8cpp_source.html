147void AnalogCircuit::CostFunctionV(double& current, double voltage) {
148 double I1 = current;
149 double J1 = 0.0;
150 double J0 = 0.0;
151 double alpha = 1.0;
152
153 do {
154 double sumVoltage = 0.0;
155 list<Component*>::iterator it;
156 for (it = component.begin(); it != component.end(); ++it) {
157 double Vcomponent = (*it)->GetVoltage(I1, T);
158 sumVoltage += Vcomponent;
159 }
160 J1 = sumVoltage - voltage;
161 //Reduce alpha if J1 and J0 are of opposite signs or are equal to each other
162 if ((abs(J0 - J1) != (J0 - J1)) || J0 == J1)alpha /= 2.0;
163
164 if (abs(J1) > tolerance) {
165 if (J1 < 0) {//increase the current
166 I1 += alpha;
167 }
168 else {
169 I1 -= alpha;
170 }
171 J0 = J1;
172 }
173 if (alpha < tolerance / 1000000.0) {//reset alpha
174 alpha = ((double)rand() / (RAND_MAX)) + 0.5;//between 0.5 and 1.5
175 }
176 } while (abs(J1) > tolerance);
177
178 fout << setw(12) << I1;
179 list<Component*>::iterator it;
180 for (it = component.begin(); it != component.end(); ++it) {
181 fout << setw(12) << (*it)->GetVoltage(I1, T);
182 ypos = (windowHeight * (*it)->GetVoltage(I1, T) / Vpeak) / 2.0 + scalingFactor * windowHeight / 2.0;
183 //xpos is the time in seconds it should be the timestep
184 xpos = accumulatedTime * windowWidth / timeMax;
185 (*it)->Display(xpos, ypos);
186 (*it)->Update(I1,voltage);
187 }
188 fout << endl;
189
190 current = I1;
191}
